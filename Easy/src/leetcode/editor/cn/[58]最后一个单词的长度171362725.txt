class Solution {
    public int lengthOfLastWord(String s) {
        
       
        
        
        int maxLength = 0;
        int perWordLength = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) != ' '){
                perWordLength++;
                maxLength = perWordLength;
            }else{
                perWordLength = 0;
            }
            
        }
        return maxLength;
    }
}
